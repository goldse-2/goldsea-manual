            // sElec: document.getElementById('summary-elec-cost'),
            sLabor: document.getElementById('summary-labor-cost'),
            sMarginal: document.getElementById('summary-marginal-cost'),
            sFixed: document.getElementById('summary-fixed-cost')
            // Removed tName and tBody references
        };
        
        // Initialize app
        function init() {
            bindEvents();
            renderProductList();
            loadProductToForm(currentProductId);
            calculateAndRender();
        }

        // Helper function for +/- buttons
        function adjustNumber(inputId, change) {
            const input = document.getElementById(inputId);
            if(!input) return;
            
            let val = parseInt(input.value) || 0;
            let min = parseInt(input.getAttribute('min'));
            let max = parseInt(input.getAttribute('max'));
            
            val += change;
            if(!isNaN(min) && val < min) val = min;
            if(!isNaN(max) && val > max) val = max;
            
            input.value = val;
            syncFormToProduct();
            // Trigger the calculation update
        }

        // Generate unique ID
        function generateId() {
            return 'prod_' + Math.random().toString(36).substr(2, 9);
        }

        // Get current product object
        function getCurrentProduct() {
            return products.find(p => p.id === currentProductId);
        }

        // Update product list UI
        function renderProductList() {
            // Update dropdown
            els.productSelector.innerHTML = '';
            products.forEach(p => {
                const opt = document.createElement('option');
                opt.value = p.id;
                opt.textContent = p.name;
                if(p.id === currentProductId) opt.selected = true;
                els.productSelector.appendChild(opt);
            });

            // Update list container
            els.productListContainer.innerHTML = '';
            products.forEach(p => {
                const div = document.createElement('div');
                const isSelected = p.id === currentProductId;
                div.className = `p-3 rounded-lg border ${isSelected ? 'border-indigo-500 bg-indigo-50' : 'border-gray-200 bg-white hover:bg-gray-50'} cursor-pointer flex justify-between items-center transition-colors`;
                
                div.innerHTML = `
                    <div class="flex-1" onclick="switchProduct('${p.id}')">
                        <div class="font-medium ${isSelected ? 'text-indigo-700' : 'text-gray-700'}">${p.name}</div>
                        <div class="text-xs text-gray-500">售价: ¥${p.price}</div>
                    </div>
                    ${products.length > 1 ? `
                        <button onclick="deleteProduct('${p.id}', event)" class="text-red-400 hover:text-red-600 p-1 rounded hover:bg-red-50" title="删除">
                            <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24" xmlns="http://www.w3.org/2000/svg"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"></path></svg>
                        </button>
                    ` : ''}
                `;
                els.productListContainer.appendChild(div);
            });
        }


        function switchProduct(id) {
            if(id === currentProductId) return;
            currentProductId = id;
            renderProductList();
            loadProductToForm(id);
            calculateAndRender();
        }

        function deleteProduct(id, event) {
            event.stopPropagation();
            if(products.length <= 1) return; // Prevent deleting last
            
            products = products.filter(p => p.id !== id);
            if(currentProductId === id) {
                currentProductId = products[0].id;
            }
            renderProductList();
            loadProductToForm(currentProductId);
            calculateAndRender();
        }

        function addNewProduct() {
            const newId = generateId();
            const newProduct = {
                id: newId,
                name: `新产品 ${products.length + 1}`,
                sales: 0,
                inventory: 0,
                price: 20.0,
                materialPriceType: '60',
                materialCustomPrice: 60,
                weight: 50,
                printTime: 3,
                packageCost: 1,
                otherMaterials: 0,
                designTime: 2,
                shippingCost: 8,
                adCost: 0,        
                returnLoss: 0     
            };
            products.push(newProduct);
            currentProductId = newId;
            renderProductList();
            loadProductToForm(currentProductId);
            calculateAndRender();
        }

        function loadProductToForm(id) {
            const p = products.find(prod => prod.id === id);
            if(!p) return;

            els.fSales.value = p.sales;
            els.fInventory.value = p.inventory;
            els.fName.value = p.name;
            els.fPrice.value = p.price;
            els.fMaterialType.value = p.materialPriceType;
            if(p.materialPriceType === 'custom') {
                els.fMaterialCustom.classList.remove('hidden');
                els.fMaterialCustom.value = p.materialCustomPrice;
            } else {
                els.fMaterialCustom.classList.add('hidden');
            }

            els.fWeight.value = p.weight;
            els.fPrintTime.value = p.printTime;
            els.fPackage.value = p.packageCost;
            els.fOtherMaterials.value = p.otherMaterials;
            els.fShipping.value = p.shippingCost;
            els.fAdCost.value = p.adCost || 0;
            // 载入数据
            els.fReturnLoss.value = p.returnLoss || 0;
            // 载入数据

            els.titleName.textContent = `参数设置 - ${p.name}`;
        }

        function syncFormToProduct() {
            const p = getCurrentProduct();
            if(!p) return;

            p.sales = parseInt(els.fSales.value) || 0;
            p.inventory = parseInt(els.fInventory.value) || 0;
            p.name = els.fName.value || '未命名产品';
            p.price = parseFloat(els.fPrice.value) || 0;
            p.materialPriceType = els.fMaterialType.value;
            p.materialCustomPrice = parseFloat(els.fMaterialCustom.value) || 0;
            p.weight = parseFloat(els.fWeight.value) || 0;
            p.printTime = parseFloat(els.fPrintTime.value) || 0;
            p.packageCost = parseFloat(els.fPackage.value) || 0;
            p.otherMaterials = parseFloat(els.fOtherMaterials.value) || 0;
            p.shippingCost = parseFloat(els.fShipping.value) || 0;
            p.adCost = parseFloat(els.fAdCost.value) || 0;             // 同步数据
            p.returnLoss = parseFloat(els.fReturnLoss.value) || 0;     // 同步数据

            // Update title and list slightly without full re-render
            els.titleName.textContent = `参数设置 - ${p.name}`;
            // Only update the selected option text and list item text
            const selectedOption = els.productSelector.querySelector(`option[value="${p.id}"]`);
            if(selectedOption) selectedOption.textContent = p.name;
            
            renderProductList(); // Re-render to update names in list easily
            calculateAndRender();
        }

        function bindEvents() {
            els.btnAddProduct.addEventListener('click', addNewProduct);
            els.productSelector.addEventListener('change', (e) => {
                switchProduct(e.target.value);
            });
            els.fMaterialType.addEventListener('change', (e) => {
                if(e.target.value === 'custom') {
                    els.fMaterialCustom.classList.remove('hidden');
                } else {
                    els.fMaterialCustom.classList.add('hidden');
                }
                syncFormToProduct();
            });
            // Add input listeners to all form fields
            const formInputs = [
                els.fSales, els.fInventory, 
                els.fName, els.fPrice, els.fMaterialCustom, els.fWeight, 
                els.fPrintTime, els.fPackage, 
                els.fOtherMaterials, els.fShipping,
                els.fAdCost, els.fReturnLoss 
            ];
            formInputs.forEach(input => {
                input.addEventListener('input', syncFormToProduct);
            });
            // Global inputs
            els.gShopCost.addEventListener('input', (e) => {
                globalSettings.shopCost = parseFloat(e.target.value) || 0;
                calculateAndRender();
            });
            // 新增囤货监听
            els.gStockCount.addEventListener('input', (e) => {
                globalSettings.stockCount = parseInt(e.target.value) || 0;
                calculateAndRender();
            });
            els.gStockPrice.addEventListener('input', (e) => {
                globalSettings.stockPrice = parseFloat(e.target.value) || 0;
                calculateAndRender();
            });
        }

        function calculateAndRender() {
            const currentP = getCurrentProduct();
            if(!currentP) return;

            // --- 1. 计算全店汇总数据 (Whole Store Totals) ---
            let shopRevenue = 0;
            let shopMarginal = 0;
            let shopCashMarginal = 0;
            let shopDesignCost = 0;
            let shopSoldCost = 0;
            let shopInvCost = 0;
            products.forEach(p => {
                let materialPricePerKg = p.materialPriceType === 'custom' ? p.materialCustomPrice : parseFloat(p.materialPriceType);
                const unitMaterialCost = (p.weight / 1000) * materialPricePerKg;
                const unitTotalMaterialCost = unitMaterialCost + p.otherMaterials;
                const unitElecCost = p.printTime * globalSettings.printerPowerKW * globalSettings.elecPrice;
                
                const unitMarginalCost = unitTotalMaterialCost + unitElecCost + p.packageCost + p.shippingCost + (p.adCost || 0) + (p.returnLoss || 0);
                const unitCashMarginalCost = unitTotalMaterialCost + unitElecCost + p.packageCost + p.shippingCost + (p.adCost || 0) + (p.returnLoss || 0);
                
                const unitInventoryHardCost = unitTotalMaterialCost + unitElecCost;

                const costFromSold = p.sales * unitMarginalCost;
                const costFromInventory = p.inventory * unitInventoryHardCost;
                const cashCostFromSold = p.sales * unitCashMarginalCost;

                shopRevenue += (p.sales * p.price);
                shopMarginal += (costFromSold + costFromInventory);
                shopCashMarginal += (cashCostFromSold + costFromInventory);
                
                shopSoldCost += costFromSold;
                shopInvCost += costFromInventory;
            });
            // 计算囤货成本
            const totalStockCost = globalSettings.stockCount * globalSettings.stockPrice;
            if(els.dStockCost) {
                els.dStockCost.textContent = `¥${totalStockCost.toFixed(2)}`;
            }

            const totalFixedCost = globalSettings.shopCost;
            // 现金流算法改变：不再按件算材料的现金流出，而是直接扣除一次性的耗材囤货费用
            // 纯利润算法不变（按件消耗算材料费）
            const dashNetProfit = shopRevenue - shopMarginal - totalFixedCost;
            // 现金流算法改变：不再按件算材料的现金流出，而是直接扣除一次性的耗材囤货费用（假设买来的材料目前就是这些）
            // 注意：因为 shopCashMarginal 里原本包含了"已用掉的材料费"，为了避免重复扣除，
            // 我们可以选择：1. 扣除囤货费后，退回单件里算掉的材料费（只算电、手工、包装运费）。
            // 或者简单粗暴地把它当作一个"额外垫资"来算：你不仅花了买这批耗材的钱，还花了做出来的材料成本（这是不合理的重复）。
            // 更合理的逻辑：
            // 真实的现金流 = 总收入 - 总开店成本 - 总囤货资金 - (扣除掉单件物料费的边际成本，即电费、包装费、运费、广告、退货等直接相关费用)
            
            let cashMarginalWithoutMaterial = 0;
            products.forEach(p => {
                const curAdCost = p.adCost || 0;
                const curReturnLoss = p.returnLoss || 0;
                // 单件中不再算入物料费(因为在总囤货里出了钱)
                const unitCashCostNoMat = (p.printTime * globalSettings.printerPowerKW * globalSettings.elecPrice) + p.packageCost + p.shippingCost + curAdCost + curReturnLoss;
                const invCashCostNoMat = (p.printTime * globalSettings.printerPowerKW * globalSettings.elecPrice); // 压货也得耗电
                
                cashMarginalWithoutMaterial += (p.sales * unitCashCostNoMat) + (p.inventory * invCashCostNoMat);
            });
            const dashCashProfit = shopRevenue - globalSettings.shopCost - totalStockCost - cashMarginalWithoutMaterial;
            // 更新全店看板 DOM
            els.dProdName.textContent = `共 ${products.length} 款产品`;
            els.dRevenue.textContent = `¥${shopRevenue.toFixed(2)}`;
            els.dTotalMarginal.textContent = `¥${shopMarginal.toFixed(2)}`;
            els.dFixedCost.textContent = `¥${totalFixedCost.toFixed(2)}`;
            
            els.tSoldCost.textContent = `¥${shopSoldCost.toFixed(2)}`;
            els.tInvCost.textContent = `¥${shopInvCost.toFixed(2)}`;
            
            els.dCashProfit.textContent = `¥${dashCashProfit.toFixed(2)}`;
            els.dCashProfit.className = `text-2xl font-bold ${dashCashProfit >= 0 ? 'text-green-600' : 'text-red-500'}`;
            // --- 2. 计算并展示【当前选中产品】的单件成本拆解 ---
            if (els.dCurrentPName) els.dCurrentPName.textContent = currentP.name;
            let curMaterialPricePerKg = currentP.materialPriceType === 'custom' ? currentP.materialCustomPrice : parseFloat(currentP.materialPriceType);
            const curUnitMaterialCost = (currentP.weight / 1000) * curMaterialPricePerKg;
            const curUnitTotalMaterialCost = curUnitMaterialCost + currentP.otherMaterials;
            const curUnitElecCost = currentP.printTime * globalSettings.printerPowerKW * globalSettings.elecPrice;
            
            const curAdCost = currentP.adCost || 0;
            const curReturnLoss = currentP.returnLoss || 0;
            const curUnitMarginalCost = curUnitTotalMaterialCost + curUnitElecCost + currentP.packageCost + currentP.shippingCost + curAdCost + curReturnLoss;
            els.dSplitMat.textContent = `¥${curUnitTotalMaterialCost.toFixed(2)}`;
            els.dSplitElec.textContent = `¥${curUnitElecCost.toFixed(2)}`;
            els.dSplitPkg.textContent = `¥${currentP.packageCost.toFixed(2)}`;
            // 更新运费展示：将原先单纯的运费，改成 运费+广告+退货 的综合展示，以防太长
            els.dSplitShip.innerHTML = `运/广/退: <span class="font-medium text-gray-700">¥${(currentP.shippingCost + curAdCost + curReturnLoss).toFixed(2)}</span>`;
            els.dSplitTotal.textContent = `¥${curUnitMarginalCost.toFixed(2)}`;
        }

        // Run setup
        window.onload = init;
